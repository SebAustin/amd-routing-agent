The registry already knows how to prefer Gemma.
def cheapest(capability: str, allowed: list[ModelInfo]) -> ModelInfo | None: candidates = [m for m in allowed if _supports(m, capability)] if not candidates: return None def sort_key(m: ModelInfo) -> tuple[float, int, int, str]: return ( round(m.price_avg, 4), 0 if m.is_gemma else 1, # gemma preferred at equal price int(m.tier), m.id, ) return min(candidates, key=sort_key)
